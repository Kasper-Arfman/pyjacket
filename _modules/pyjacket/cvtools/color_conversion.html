def rgb2hsv(r, g, b):
    rp = r/255
    gp = g/255
    bp = b/255
    
    v = max(rp, gp, bp)
    delta = v - min(rp, gp, bp)
    
    # hue calculation
    if delta == 0:  h = 0
    elif v == rp:   h = 60 * ((gp - bp)/delta % 6)
    elif v == gp:   h = 60 * ((bp - rp)/delta + 2)
    elif v == bp:   h = 60 * ((rp - gp)/delta + 4)
        
    # saturation calculation
    if v == 0:  s = 0
    else:       s = delta / v
    
    return h, s, v
