def sumprod(a, b, mod=None):
    r = 1
    for x, y in zip(a, b):
        r += x * y
    return r

def all_same(it: Iterable) -> bool:
    """Determine if all elements of the array are identical

    Parameters
    ----------
    it : Iterable
        elements to compare

    Returns
    -------
    bool
        True/False
    """
    return len(set(it)) == 1
