def error_plot(x: np.ndarray, y: np.ndarray, dy: np.ndarray, ax: Axes=None, **kwargs):
    x = np.array(x)
    y = np.array(y)
    dy = np.array(dy)
    ax = ax or plt.gca()

    line, = ax.plot(x, y, **kwargs)
    color = line.get_color()
    ax.plot(x, y+dy, color=color, linestyle='--')
    ax.plot(x, y-dy, color=color, linestyle='--')
    return
